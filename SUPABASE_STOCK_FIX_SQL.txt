-- 1. Ajoute la colonne 'manage_stock' à la table 'products' si elle n'existe pas
-- Par défaut = TRUE (Mode Strict)
ALTER TABLE public.products 
ADD COLUMN IF NOT EXISTS manage_stock BOOLEAN DEFAULT TRUE;

-- 2. Met à jour tous les produits existants pour qu'ils soient en mode STRICT par défaut
-- (Ceci assure que tout est propre)
UPDATE public.products 
SET manage_stock = TRUE 
WHERE manage_stock IS NULL;
